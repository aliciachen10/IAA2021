Question 1

proc sql;
	select employee_name, int(('1SEP2018'd - Employee_Hire_Date)/365.25) as YOS
	from jupiter.employee_addresses a, jupiter.employee_payroll p
	where a.employee_id = p.employee_id ;
quit;


Question 2
-----
proc sql;
	select p.product_id, p.product_name, sum(quantity) as Total_Sold
	from jupiter.product_dim p, jupiter.order_fact o
	where p.product_id = o.product_id and year(Order_Date) >= 2010 and Product_Name = 'Smasher Shorts'
	group by 1,2
	order by 3 desc, 2;
quit;
