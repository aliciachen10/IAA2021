proc sql;
select 'Total Paid to All Female Sales Representatives', sum(Salary), count(*) as total
 from jupiter.salesstaff
 where gender='F' and Job_Title like '%Rep%'
union
select 'Total Paid to All Male Sales Representatives', sum(Salary), count(*) as total 
 from jupiter.salesstaff
 where gender='M' and Job_Title like '%Rep%';
quit;
